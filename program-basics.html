2. Add Two Numbers

#include <stdio.h>

int main()
{
    int a,b,sum;

    printf("Enter two numbers: ");
    scanf("%d %d",&a,&b);

    sum=a+b;

    printf("Sum = %d",sum);

    return 0;
}

3. Swap Two Numbers

#include <stdio.h>

int main()
{
    int a,b,temp;

    printf("Enter values: ");
    scanf("%d %d",&a,&b);

    temp=a;
    a=b;
    b=temp;

    printf("After swapping: %d %d",a,b);

    return 0;
}

4. Find Area of Circle

#include <stdio.h>

int main()
{
    float radius,area;

    printf("Enter radius: ");
    scanf("%f",&radius);

    area=3.14*radius*radius;

    printf("Area = %.2f",area);

    return 0;
}

5. Simple Calculator

#include <stdio.h>

int main()
{
    int a,b;

    char op;

    printf("Enter operator (+,-,*,/): ");
    scanf("%c",&op);


    printf("Enter two numbers: ");
    scanf("%d %d",&a,&b);


    switch(op)
    {
        case '+':
            printf("%d",a+b);
            break;

        case '-':
            printf("%d",a-b);
            break;

        case '*':
            printf("%d",a*b);
            break;

        case '/':
            printf("%d",a/b);
            break;

        default:
            printf("Invalid Operator");
    }


    return 0;
}
