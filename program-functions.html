Chapter 7 - Functions in C C Functions Programs
C Functions Programs

1. Function Without Arguments

#include <stdio.h>


void hello()
{
    printf("Hello From Function");
}


int main()
{

    hello();

    return 0;

}

2. Function With Arguments

#include <stdio.h>


void add(int a,int b)
{
    printf("Sum = %d",a+b);
}



int main()
{

    int x=5,y=10;


    add(x,y);


    return 0;

}

3. Function With Return Value

#include <stdio.h>


int square(int n)
{
    return n*n;
}



int main()
{

    int result;


    result=square(5);


    printf("Square = %d",result);



    return 0;

}

4. Call By Value

#include <stdio.h>


void change(int x)
{
    x=100;
}



int main()
{

    int a=10;


    change(a);


    printf("%d",a);



    return 0;

}

5. Call By Reference Using Pointer

#include <stdio.h>


void change(int *x)
{
    *x=100;
}



int main()
{

    int a=10;


    change(&a);


    printf("%d",a);



    return 0;

}

6. Factorial Using Recursion

#include <stdio.h>


int factorial(int n)
{

    if(n==0)
        return 1;


    return n * factorial(n-1);

}



int main()
{

    int num=5;


    printf("Factorial = %d",
    factorial(num));


    return 0;

}
