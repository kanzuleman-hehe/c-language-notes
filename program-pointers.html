C Pointers Programs

1. Find Address of Variable

#include <stdio.h>


int main()
{

    int num=10;


    printf("Value = %d\n",num);


    printf("Address = %p",&num);



    return 0;

}

2. Swap Two Numbers Using Pointer

#include <stdio.h>


void swap(int *a,int *b)
{

    int temp;


    temp=*a;

    *a=*b;

    *b=temp;

}



int main()
{

    int x=5,y=10;


    swap(&x,&y);



    printf("x = %d y = %d",x,y);



    return 0;

}

3. Access Array Using Pointer

#include <stdio.h>


int main()
{

    int arr[5]={10,20,30,40,50};


    int *ptr;


    ptr=arr;



    for(int i=0;i<5;i++)
    {

        printf("%d\n",*(ptr+i));

    }



    return 0;

}

4. Pointer Arithmetic Example

#include <stdio.h>


int main()
{

    int a=10;


    int *ptr=&a;



    printf("%d\n",*ptr);



    ptr++;



    printf("Pointer moved");



    return 0;

}

5. Dynamic Memory Allocation Using malloc()

#include <stdio.h>
#include <stdlib.h>


int main()
{

    int *ptr;


    ptr=(int*)malloc(sizeof(int));



    *ptr=50;



    printf("Value = %d",*ptr);



    free(ptr);



    return 0;

}
