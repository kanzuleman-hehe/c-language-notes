1. Student Record Using Structure

#include <stdio.h>


struct Student
{
    int roll;
    char name[20];
    float marks;
};



int main()
{

    struct Student s;



    printf("Enter Roll No: ");
    scanf("%d",&s.roll);


    printf("Enter Name: ");
    scanf("%s",s.name);


    printf("Enter Marks: ");
    scanf("%f",&s.marks);



    printf("\nStudent Details\n");

    printf("Roll: %d\n",s.roll);

    printf("Name: %s\n",s.name);

    printf("Marks: %.2f",s.marks);



    return 0;

}

2. Employee Record Using Structure

#include <stdio.h>


struct Employee
{

    int id;

    char name[20];

    float salary;

};



int main()

    scanf("%f",&e.salary);

    printf("ID: %d\n",e.id);

    printf("Name: %s\n",e.name);

    printf("Salary: %.2f",e.salary);



    return 0;

}

3. Array of Structures

#include <stdio.h>


struct Student
{

    int roll;

    float marks;

};



int main()
{

    struct Student s[3];


    int i;



    for(i=0;i<3;i++)
    {

        printf("Enter Roll: ");

        scanf("%d",&s[i].roll);



        printf("Enter Marks: ");

        scanf("%f",&s[i].marks);

    }




    printf("\nRecords\n");



    for(i=0;i<3;i++)
    {

        printf("%d %.2f\n",
        s[i].roll,
        s[i].marks);

    }



    return 0;

}

4. Nested Structure Example

#include <stdio.h>


struct Address
{

    char city[20];

};



struct Student
{

    int roll;

    struct Address add;

};



int main()
{

    struct Student s;


    s.roll=1;


    printf("Enter City: ");

    scanf("%s",s.add.city);



    printf("Roll: %d\n",s.roll);


    printf("City: %s",s.add.city);



    return 0;

}

5. Structure With Function

#include <stdio.h>


struct Student
{

    int roll;

    float marks;

};



void display(struct Student s)
{

    printf("Roll = %d\n",s.roll);

    printf("Marks = %.2f",s.marks);

}



int main()
{

    struct Student s={1,90.5};



    display(s);



    return 0;

}
